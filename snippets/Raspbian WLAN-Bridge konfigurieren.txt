### Notwendige Pakete installieren
sudo apt-get install hostapd bridge-utils

### Interfaces der Bridge aus der DHCP-Konfiguration rausnehmen
cat << EOF | sudo tee -a /etc/dhcpcd.conf
denyinterfaces eth0
denyinterfaces wlan0
EOF

### Bridge erzeugen
sudo brctl addbr br0

### Interfaces verbinden
sudo brctl addif br0 eth0

### Konfiguration der Bridge hinzuf端gen
cat << EOF | sudo tee -a /etc/network/interfaces
# Bridge setup
auto br0
iface br0 inet manual
bridge_ports eth0 wlan0
EOF

### WLAN-Konfiguration f端r hostapd erzeugen
cat << EOF | sudo tee /etc/hostapd/hostapd.conf
# Schnittstelle und Treiber
interface=wlan0
#driver=nl80211

# Bridge
bridge=br0

# WLAN-Konfiguration
ssid=Hades
channel=1
hw_mode=g
ieee80211n=1            # 802.11n support
ieee80211d=1            # limit the frequencies used to those allowed in the country
country_code=DE         # country code
wmm_enabled=1           # QoS support

# WLAN-Verschl端sselung
auth_algs=1             # 1=WPA, 2=WEP, 3=both
wpa=2                   # WPA2 only
wpa_key_mgmt=WPA-PSK
rsn_pairwise=CCMP
wpa_passphrase=PkThcbyiJWYZ8CydVQya
EOF

### Die erzeugte Datei als Default-Kobfiguration f端r hostapd setzen
sudo sed -i '/.*#DAEMON_CONF.*/c\DAEMON_CONF="/etc/hostapd/hostapd.conf"' /etc/default/hostapd

### hostapd auf Autostart setzen
sudo systemctl unmask hostapd
sudo systemctl enable hostapd

### edit /etc/sysctl.conf and set the ipv4/ipv6 settings
sudo nano /etc/sysctl.conf

net.ipv4.ip_forward=1
net.ipv6.conf.all.disable_ipv6 = 1


### Reboot
sudo reboot
